CREATE TABLE carreras (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  carrera varchar(200) NOT NULL,
  PRIMARY KEY (id)
) ENGINE = InnoDB;


CREATE TABLE materias (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  materia varchar(200) NOT NULL,
  creditos int UNSIGNED NOT NULL,
  semestre int UNSIGNED NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY (materia)
) ENGINE = InnoDB;


CREATE TABLE alumnos (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  matricula char(8) NOT NULL,
  nombre1 varchar(50) NOT NULL,
  nombre2 varchar(50),
  apellido1 varchar(50) NOT NULL,
  apellido2 varchar(50),
  telefono char(10) NOT NULL,
  correo varchar(100) NOT NULL,
  semestre int UNSIGNED NOT NULL DEFAULT '1',
  id_carrera int UNSIGNED NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY (telefono),
  UNIQUE KEY (correo),
  UNIQUE KEY (matricula),
  FOREIGN KEY(id_carrera) REFERENCES carreras(id)
) ENGINE = InnoDB;


CREATE TABLE cursos (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  titulo varchar(300) NOT NULL,
  link varchar(300) NOT NULL,
  unidad INT UNSIGNED NOT NULL,
  id_materia int UNSIGNED NOT NULL,
  PRIMARY KEY (id),
  FOREIGN KEY (id_materia) REFERENCES materias(id)
) ENGINE = InnoDB;


CREATE TABLE profesores (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  clave char(8) NOT NULL,
  nombre1 varchar(50) NOT NULL,
  nombre2 varchar(50),
  apellido1 varchar(50) NOT NULL,
  apellido2 varchar(50),
  telefono char(10) NOT NULL,
  correo varchar(100) NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY (telefono),
  UNIQUE KEY (clave),
  UNIQUE KEY (correo)
) ENGINE = InnoDB;


CREATE TABLE alumnos_materias (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  id_alumno int UNSIGNED NOT NULL,
  id_materia int UNSIGNED NOT NULL,
  califa int UNSIGNED NOT NULL, 
  PRIMARY KEY (id),
  FOREIGN KEY (id_alumno) REFERENCES alumnos(id),
  FOREIGN KEY (id_materia) REFERENCES materias(id)
) ENGINE = InnoDB;


CREATE TABLE materias_carreras (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,
  id_materia int UNSIGNED NOT NULL,
  id_carrera int UNSIGNED NOT NULL,  
  PRIMARY KEY (id),
  FOREIGN KEY (id_carrera) REFERENCES carreras(id),
  FOREIGN KEY (id_materia) REFERENCES materias(id)
) ENGINE = InnoDB;


CREATE TABLE materias_profesores (
  id int UNSIGNED NOT NULL AUTO_INCREMENT,  
  id_materia int UNSIGNED NOT NULL,
  id_profesor int UNSIGNED NOT NULL,
  PRIMARY KEY (id),
  FOREIGN KEY (id_profesor) REFERENCES profesores(id),
  FOREIGN KEY (id_materia) REFERENCES materias(id)
) ENGINE = InnoDB;
